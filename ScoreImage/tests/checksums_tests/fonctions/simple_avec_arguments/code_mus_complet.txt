# Ce fichier permet de calculer les notes et les 
# durées pour les statistiques.
\relative c'' { r8 g16 c e g, c e r8 g,16 c e g, c e }
<< r16 e8.~ e4 r16 e8.~ e4 // c2 c2 >>
\relative c'' { r8 a16 d f a, d f r8 a,16 d f a, d f }
<< r16 d8.~ d4 r16 d8.~ d4 // c2 c2 >>
\relative c'' { r8 g16 d’ f g, d’ f r8 g,16 d’ f g, d’ f }
<< r16 d8.~ d4 r16 d8.~ d4 // b2 b2 >>
